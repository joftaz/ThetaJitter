%*********************************************************************
% Theory of the Hilbert Spectrum: 
% Code Examples for Reproducible Research
%*********************************************************************
%
% Description: This package consist of several files for performing 
%              Hilbert Spectral Analysis (HSA) as presented in the 
%              following paper:
%
%       @article{HSA2015,
%           title={Theory of the Hilbert Spectrum},
%           author={Sandoval, S. and De~Leon, P.~L.~},
%           journal={{Applied and Computational Harmonic Analysis}},
%           year = {\noop{2015}in review},  }
%
%
%==================================================================
%
%   Below is a summary of the contents in this package:
%
%----------------------------------------------------------
%   0_DocumentationAndOtherFiles - Documetation and other files
%----------------------------------------------------------
%
%----------------------------------------------------------
%   1_VisualizationExamples - Demos for visualizing the Hilbert Spectrum
%----------------------------------------------------------
%
%       1_BasicComponents
%           Demo_1_SimpleHarmicComponent - Visualization for the Simple Harmonic Component
%           Demo_2_SimpleAMComponent     - Visualization for the Simple Amplitude Modulated (AM) Component
%           Demo_3_SimpleFMComponent     - Visualization for the Simple Frequency Modulated (FM) Component
%           Demo_4_AMFMComponent         - Visualization for the AM--FM Component
%
%       2_WidebandComponents
%           Demo_1_WideBandAMComponent - Visualization for an AM--FM Component with wideband AM; Note the harmonic structure in the Fourier spectrum
%           Demo_2_WideBandFMComponent - Visualization for an AM--FM Component with wideband FM; Note the harmonic structure in the Fourier spectrum
%
%       3_otherExamples
%           Demo_1_MultiComponentSignal - Visualization for a multicomponent AM--FM Signal
%           Demo_2_ExportPNG            - Exporting an Visual
%           Demo_3_BeatEffect           - Visualization for the `beat effect' 
%
%       4_Triangle&SinusoidalFM
%           Demo_1_TriangleWave  - Visualization for the periodic triangular waveform 
%           Demo_2_Sinusiodal_FM - Visualization for the sinusoidal FM signal 
%
%----------------------------------------------------------
%   2_DecompositionExamples - HSA Examples
%----------------------------------------------------------
%       1_SiftingAlgorithm
%           Demo_1_SiftingAlgorithm - Visualization for a single iteration of sifting
%           
%       2_SyntheticSignals
%           Demo_1_slowAMfastFMSignal - HSA example for the slow-AM, fast-FM signal example in Part III 
%           Demo_2_fastAMslowFMSignal - HSA example for the fast-AM, slow-FM signal example in Part III
%
%       3_RealWorldSignals
%           Demo_1_Cello  - HSA example for the Cello signal example in Part III 
%           Demo_2_Speech - HSA example for the Speech signal example in Part III 
%
%----------------------------------------------------------
%   3_RequiredFunctions - Supporting functions
%----------------------------------------------------------
%
%           myReqFuncs    - files developed for this research
%           otherReqFuncs - files available under BSL licences
%
%==================================================================
